    %% 1. QUÉT ĐƠN & CẢNH BÁO QUÁ KÝ %%
    subgraph "BUOC 1: QUET BARCODE VA CANH BAO QUA KY THONG MINH"
        Start["Nhan vien quet ma don, Nhap so ky thuc te"] --> CompareWeight{"So sanh can nang khai bao"}

 Bat buoc chup 3 anh va 1 video minh chung"]
        CompareWeight -->|Binh thuong hoac Trong pham vi cho phep| NormalProceed["Cho phep luu truc tiep 
 Khuyen khich chup anh bang chung"]
        
        AlertRed --> SmartCompress["NEN DUNG LUONG SIEU NHO 
 - 3 Ảnh nén JPEG: 9MB giam con 450 KB 
 - 1 Video 480p nén: 10MB giam con 1 MB 
 - Tong 1 don hang chi nang 1.5 MB

        NormalProceed --> SmartCompress
    end

    %% 2. CHIA CA 24H LIÊN TỤC & RESET NGÀY MỚI %%
    subgraph "BUOC 2: CHU KY LAM VIEC 24H KHEP KIN VA TU DONG CHIA CA"
        SmartCompress --> DateCalc{"Loc Ngay lam viec thuc te"}
        
        DateCalc -->|Quet luc 00:00 den 07:59:59 Sang| PrevDay["Ngay lam viec = Ngay hien tai tru 1 Ngay"]
        DateCalc -->|Quet luc 08:00 den 23:59:59 Dem| CurrDay["Ngay lam viec = Ngay hien tai"]
        
        PrevDay --> ShiftFilter{"Xac dinh Ca"}
        CurrDay --> ShiftFilter
        
        ShiftFilter -->|Gio quet tu 08:00:00 den 16:59:59| ShiftDay["CA SANG 
 Cong vao bo dem Ca Sáng thuc te"]
        ShiftFilter -->|Gio quet tu 17:00:00 den 07:59:59 sang mai| ShiftNight["CA DEM 
 Cong vao bo dem Ca Dem thuc te"]
        
        ShiftDay --> ActiveScreen["Hien thi so luong thuc te da xo tren Home Screen"]
        ShiftNight --> ActiveScreen
        
        ActiveScreen --> TimeCheck{"Dung moc 08:00 Sang"}
        
        TimeCheck -->|Dung 08:00 Sang ngay moi| HandoverModal["HIEN THI HOP THOAI BAN GIAO CA 
 - Tong ket san luong ngay cu Ca Sang X Ca Dem Y 
 - Kiem tra trang thai dong bo du lieu 
 - Nhan vien ky bien ban doi soat"]
        HandoverModal -->|Bam Xac nhan va bat dau ca moi| ResetAction["App tu dong Reset bo dem ve 0 don 
 Dong goi lich su ngay cu vao DB trung tam"]
        TimeCheck -->|Dang trong ca lam viec| NetworkRoute["Truyen tai du lieu"]
    end

    %% 3. ĐƯỜNG TRUYỀN BẢO MẬT API KEY (3 KÊNH THÔNG MINH) %%
    subgraph "BUOC 3: BA KENH TRUYEN BAO MAT VA CHONG TRUY CAP LAU"
        NetworkRoute --> CheckServer{"Server Windows Online hay Offline"}
        
        %% KÊNH 1: SERVER ONLINE %%
        CheckServer -->|Server ONLINE Ca Sang| VerifyKey1{"Cua ngo API Key"}
        VerifyKey1 -->|API Key hop le| UploadLocal["Gui ve may Windows noi bo kho 
 Mang LAN sieu toc duoi 0.5 giay tren don"]
        VerifyKey1 -->|Thiet bi la ket noi Wi-Fi| BlockHacker["Chan truy cap - Loi 401"]
        
        UploadLocal --> DriveSyncNgam["Server tu dong day len Google Drive cuon chieu"]
        UploadLocal --> CleanLocal1["App xoa file tam tren dien thoai ngay lap tuc 
 Bo nho chiem dung vinh vien = 0 MB"]

        %% KÊNH 2: SERVER OFFLINE - CÓ INTERNET %%
        CheckServer -->|Server OFFLINE Ca Dem| CheckInternet{"Dien thoai co Internet Wi-Fi hoac 4G"}
        CheckInternet -->|CO INTERNET| DirectDrive["DAY THANG LEN GOOGLE DRIVE 
 Dien thoai tu dong tai file len dam may"]
        DirectDrive --> CleanLocal2["App xoa file tam tren dien thoai ngay lap tuc 
 Bo nho chiem dung vinh vien = 0 MB"]

        %% KÊNH 3: SERVER OFFLINE - MẤT MẠNG %%
        CheckInternet -->|MAT INTERNET| SandboxSave["LUU TAM VAO APP SANDBOX 
 Luu an bao mat trong app, dung luong sieu nghe 1.5MB tren don 
 Ca ca dem 300 don chi ton khoang 450 MB"]
        SandboxSave --> MarkYellow["Hien thi trang thai Cho dong bo"]
        
        MarkYellow --> ServerBackOn["Khi Server Windows duoc bat lai luc 8h sang"]
        ServerBackOn --> AutoSyncQueue["Tu dong dong bo ngam toan bo hang doi"]
        AutoSyncQueue --> VerifyKey2{"Kiem tra API Key"}
        VerifyKey2 -->|API Key hop le| UploadServer2["Gui ve may Windows noi bo kho"]
        UploadServer2 --> CleanLocal3["App xoa sach file tam ca dem tren may"]
    end

    %% 4. BÁO CÁO VÀ TRA CỨU LỊCH SỬ %%
    subgraph "BUOC 4: BAO CAO LICH SU VA DOI SOAT THONG TIN"
        DriveSyncNgam --> HistoryReport["TAB BAO CAO LICH SU NGAY 
 - Tra cuu san luong ca Sang hoac ca Dem cua bat ky ngay nao 
 - Go hoac quet ma don cu de xem lai can nang va anh bang chung 
 - Trang thai dong bo Google Drive: 100%"]
        HistoryReport --> EndFlow["KET THUC QUY TRINH AN TOAN 
 He thong bao mat tuyet doi - Khong lo day bo nho"]
        CleanLocal3 --> EndFlow
    end
